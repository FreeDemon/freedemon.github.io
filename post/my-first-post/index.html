def computeCost(X, y, theata):
    m = len(y)
    result = np.sum((np.dot(X, theata) - y)**2)/(2*m)
    return result

def gradientDescent(X, y, theata, alpha, num_iters):
    m = len(y)
    J_history = np.zeros((num_iters, 1))
    for i in range(num_iters):
        theata = theata - alpha * np.dot(X.T, (np.dot(X, theata) -y )) / m;
        print("%f-----%f" % (theata[0][0], theata[1][0])  )
        J_history[i] = computeCost(X, y, theata);
    return theata, J_history
